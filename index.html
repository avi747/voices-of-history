// ============================================================ // VOICES OF HISTORY — GAS Backend // Sheet ID: 1uxKhmVqt7rI_YH4MhL4go600EsevI-91HYDUYne8Xm4 // Tabs: Personas, Conversations // ============================================================ var SHEET_ID = "1uxKhmVqt7rI_YH4MhL4go600EsevI-91HYDUYne8Xm4"; var PERSONAS_TAB = "Personas"; var LOG_TAB = "Conversations"; var CLAUDE_API_URL = "https://api.anthropic.com/v1/messages"; var ELEVENLABS_API_URL = "https://api.elevenlabs.io/v1/text-to-speech/"; var CLAUDE_MODEL = "claude-sonnet-4-20250514"; var ELEVENLABS_MODEL = "eleven_multilingual_v2"; var MAX_TOKENS = 1000; // ============================================================ // ENTRY POINTS // ============================================================ function doGet(e) { var action = e.parameter.action; if (action === "personas") { return handleGetPersonas(); } if (action === "chat") { var body = { personaId: e.parameter.personaId, language: e.parameter.language || "en", audio: e.parameter.audio === "true", messages: JSON.parse(decodeURIComponent(e.parameter.messages || "[]")) }; return handleChat(body); } return makeResponse({ error: "Unknown action" }); } function doPost(e) { var body; try { body = JSON.parse(e.postData.contents); } catch (err) { return makeResponse({ error: "Invalid JSON body" }); } if (body.action === "chat") { return handleChat(body); } return makeResponse({ error: "Unknown action" }); } // ============================================================ // GET PERSONAS // ============================================================ function handleGetPersonas() { try { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(PERSONAS_TAB); var data = sheet.getDataRange().getValues(); if (data.length < 2) { return makeResponse({ personas: [] }); } var headers = data[0]; var personas = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var persona = {}; for (var j = 0; j < headers.length; j++) { persona[headers[j]] = row[j]; } var isActive = persona.Active === true || persona.Active === "TRUE" || persona.Active === "Yes"; if (isActive) { persona.SuggestedQ_EN = parsePiped(persona.SuggestedQ_EN); persona.SuggestedQ_HE = parsePiped(persona.SuggestedQ_HE); persona.SuggestedQ_FR = parsePiped(persona.SuggestedQ_FR); persona.SuggestedQ_RU = parsePiped(persona.SuggestedQ_RU); persona.SuggestedQ_ES = parsePiped(persona.SuggestedQ_ES); personas.push(persona); } } return makeResponse({ personas: personas }); } catch (err) { logError("handleGetPersonas", err); return makeResponse({ error: "Failed to load personas: " + err.message }); } } // ============================================================ // CHAT — calls Claude, optionally calls ElevenLabs // ============================================================ function handleChat(body) { var personaId = body.personaId; var messages = body.messages; var language = body.language || "en"; var wantAudio = body.audio === true; if (!personaId || !messages || !messages.length) { return makeResponse({ error: "Missing personaId or messages" }); } var anthropicKey = PropertiesService.getScriptProperties().getProperty("ANTHROPIC_API_KEY"); if (!anthropicKey) { return makeResponse({ error: "Anthropic API key not configured" }); } var personaRow = getPersonaRow(personaId); if (!personaRow) { return makeResponse({ error: "Persona not found: " + personaId }); } var systemPrompt = personaRow.SystemPrompt + "\n\n" + getLanguageInstruction(language); // ── Call Claude ── var claudePayload = { model: CLAUDE_MODEL, max_tokens: MAX_TOKENS, system: systemPrompt, messages: messages }; var claudeOptions = { method: "post", contentType: "application/json", headers: { "x-api-key": anthropicKey, "anthropic-version": "2023-06-01" }, payload: JSON.stringify(claudePayload), muteHttpExceptions: true }; try { var claudeResp = UrlFetchApp.fetch(CLAUDE_API_URL, claudeOptions); if (claudeResp.getResponseCode() !== 200) { logError("handleChat", "Claude " + claudeResp.getResponseCode() + ": " + claudeResp.getContentText()); return makeResponse({ error: "Claude API error" }); } var claudeData = JSON.parse(claudeResp.getContentText()); var reply = claudeData.content[0].text; logConversation(personaId, messages[messages.length - 1].content, reply, language); // ── Optionally call ElevenLabs ── var audioB64 = null; if (wantAudio) { audioB64 = getElevenLabsAudio(reply, personaRow.VoiceID, language); } var result = { reply: reply }; if (audioB64) { result.audio = audioB64; } return makeResponse(result); } catch (err) { logError("handleChat", err); return makeResponse({ error: "Request failed: " + err.message }); } } // ============================================================ // ELEVENLABS TTS // Returns base64-encoded MP3 or null on failure // ============================================================ function getElevenLabsAudio(text, voiceId, language) { if (!voiceId) return null; var elKey = PropertiesService.getScriptProperties().getProperty("ELEVENLABS_API_KEY"); if (!elKey) return null; // Trim text to 2500 chars max to control cost and latency var trimmed = text.length > 2500 ? text.substring(0, 2497) + "..." : text; var payload = { text: trimmed, model_id: ELEVENLABS_MODEL, voice_settings: { stability: 0.5, similarity_boost: 0.75, style: 0.3, use_speaker_boost: true } }; var options = { method: "post", contentType: "application/json", headers: { "xi-api-key": elKey }, payload: JSON.stringify(payload), muteHttpExceptions: true }; try { var resp = UrlFetchApp.fetch(ELEVENLABS_API_URL + voiceId, options); if (resp.getResponseCode() !== 200) { logError("getElevenLabsAudio", "ElevenLabs " + resp.getResponseCode() + ": " + resp.getContentText().substring(0, 200)); return null; } var bytes = resp.getContent(); return Utilities.base64Encode(bytes); } catch (err) { logError("getElevenLabsAudio", err); return null; } } // ============================================================ // HELPERS // ============================================================ function getPersonaRow(personaId) { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(PERSONAS_TAB); var data = sheet.getDataRange().getValues(); var headers = data[0]; var idCol = headers.indexOf("PersonaID"); var activeCol = headers.indexOf("Active"); for (var i = 1; i < data.length; i++) { var row = data[i]; var isActive = row[activeCol] === true || row[activeCol] === "TRUE" || row[activeCol] === "Yes"; if (row[idCol] === personaId && isActive) { var obj = {}; for (var j = 0; j < headers.length; j++) { obj[headers[j]] = row[j]; } return obj; } } return null; } function getLanguageInstruction(lang) { var instructions = { "en": "Respond in English.", "he": "השב בעברית. כתוב בעברית תקינה ומלאה.", "fr": "Réponds en français.", "ru": "Отвечай на русском языке.", "es": "Responde en español." }; return instructions[lang] || "Respond in English."; } function parsePiped(val) { if (!val || val === "") return []; return String(val).split("|").map(function(s) { return s.trim(); }).filter(function(s) { return s.length > 0; }); } function logConversation(personaId, userMsg, personaReply, language) { try { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(LOG_TAB); sheet.appendRow([new Date(), personaId, userMsg, personaReply, language]); } catch (err) { logError("logConversation", err); } } function logError(context, err) { try { Logger.log("[ERROR] " + context + ": " + (err.message || err)); } catch(e) {} } function makeResponse(data) { var output = ContentService.createTextOutput(JSON.stringify(data)); output.setMimeType(ContentService.MimeType.JSON); return output; } // ============================================================ // TEST FUNCTIONS // ============================================================ function testGetPersonas() { var result = handleGetPersonas(); Logger.log(result.getContent()); } function testChat() { var result = handleChat({ action: "chat", personaId: "herzl", language: "en", audio: false, messages: [{ role: "user", content: "What drove you to create the Zionist movement?" }] }); Logger.log(result.getContent()); } function testChatWithAudio() { var result = handleChat({ action: "chat", personaId: "herzl", language: "en", audio: true, messages: [{ role: "user", content: "Tell me briefly about Basel." }] });
